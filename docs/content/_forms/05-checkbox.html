Для стилизации checkbox используется следующая структура.

{% highlight HTML %}
{% endhighlight %}

PHP через расширенный ActiveForm

{% highlight PHP %} use app\widgets\ActiveForm; field($model, 'onlyVehicles')->checkbox() ?> {% endhighlight %}

или PHP вариант template

{% highlight PHP %} use \yii\bootstrap\ActiveForm; field( $model, 'onlyVehicles', [ 'options' => ['class' => 'checkbox-item'], 'template' => "{input}{beginLabel}{labelTitle}{endLabel}{error}" ]) ->checkbox(['class' => ''], false) ->label(null, ['class' => 'checkbox']); ?> {% endhighlight %}
